def dénivelé(altitudes):
    pos, neg = 0, 0
    for i in range(1, len(altitudes)):
        diff = altitudes[i] - altitudes[i - 1]
        if diff > 0:
            pos += diff
        elif diff < 0:
            neg += abs(diff)
    return (pos, neg)
Q5 — Fonction asc_desc : 
def asc_desc(trace):
    asc, desc, plat = 0, 0, 0
    for i in range(1, len(trace)):
        dt   = trace[i]['tsp'] - trace[i-1]['tsp']
        diff = trace[i]['alt'] - trace[i-1]['alt']
        if   diff > 0:  asc  += dt
        elif diff < 0:  desc += dt
        else:           plat += dt
    return (asc, desc, plat)
